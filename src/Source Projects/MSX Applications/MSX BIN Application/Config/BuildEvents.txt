;----------------------------------------------------------
;	BuildEvents.txt - by Danilo Angelo, 2021
;
;	Actions to be taken during the Build Process
;	Example: BUILD_START_ACTION	MyStartScript [PROFILE]
;----------------------------------------------------------

;----------------------------------------------------------
;	Always executed in build start
BUILD_START_ACTION [MSX_DEV_PATH]\..\..\..\a

;----------------------------------------------------------
;	Not executed if build clean without all
BEFORE_COMPILE_ACTION [MSX_DEV_PATH]\..\..\..\b
AFTER_COMPILE_ACTION [MSX_DEV_PATH]\..\..\..\c
AFTER_BINARY_ACTION [MSX_DEV_PATH]\..\..\..\d

;----------------------------------------------------------
; Always executed in build end
BUILD_END_ACTION [MSX_DEV_PATH]\..\..\..\e
